function Fruit({ type, name, isRipe, calories }) {
  this.type = type;
  this.name = name;
  this.isRipe = isRipe;
  this.calories = calories;
}

Fruit.prototype.ship = function(destination) {
  console.log(`shipping ${this.name} to ${destination}`);
};

// raspberry -----------

function Raspberry(options) {
  Fruit.call(this, options);
  this.doBearsLikeIt = options.doBearsLikeIt;
}

Raspberry.prototype = Object.create(Fruit.prototype);

Raspberry.prototype.foo = function() {
  console.log('foo!');
};

// pineapple -----------

function Pineapple(options) {
  Fruit.call(this, options);
  this.fromHawaii = options.fromHawaii;
}

Pineapple.prototype = Object.create(Fruit.prototype);

// instantiation -----------
const myRaspberry = new Raspberry({
  type: 'berry',
  name: 'raspberry',
  isRipe: true,
  calories: 200,
  doBearsLikeIt: true,
});

const myPineapple = new Pineapple({
  type: 'bush thing',
  name: 'pineapple',
  isRipe: true,
  calories: 1500,
  fromHawaii: true,
});

class Fruit {
  constructor(options) {
    this.type = options.type;
    this.name = options.name;
    this.isRipe = options.isRipe;
    this.calories = options.calories;
  }
  
  ship(destination) {
    console.log(`shipping ${this.name} to ${destination}`);
  }
}

// raspberry -----------

class Raspberry extends Fruit { // connects prototype
  constructor(options) {
    super(options); // Fruit.call
    this.doBearsLikeIt = options.doBearsLikeIt;
  }
  
  foo() {
    console.log('foo!');
  }
}

  }
}
